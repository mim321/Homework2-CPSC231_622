void main (){
   
  // วนลูปแสดงเลข 5 4 3 2 1 โดยตรวจสอบเงื่อนไขก่อน
  var x = 5;
  while (x > 0){ // ตรวจสอบเงื่อนไขก่อนทำคำสั่ง
    print(x);
    x--;
  }
   
  // วนลูปแสดงเลข 5 4 3 2 1 โดยตรวจสอบเงื่อนไขก่อน  
  do { // ทำคำสั่งก่อน
    print(x); 
    x--;    
  } while (x > 0); // แล้วตรวจสอบเงื่อนไข
   
}
